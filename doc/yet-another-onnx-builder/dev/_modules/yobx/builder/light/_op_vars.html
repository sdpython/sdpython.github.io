def _complete_ops_vars() -> None:
    binary_ops = [
        "Add",
        "And",
        "BitwiseAnd",
        "BitwiseOr",
        "BitwiseXor",
        "CastLike",
        "Div",
        "Equal",
        "Expand",
        "GatherND",
        "Greater",
        "GreaterOrEqual",
        "Less",
        "LessOrEqual",
        "MatMul",
        "Mul",
        "Or",
        "PRelu",
        "Pow",
        "Reshape",
        "StringConcat",
        "Sub",
        "Tile",
        "Unsqueeze",
        "Xor",
    ]
    for name in binary_ops:
        if hasattr(OpsVars, name):
            continue
        setattr(
            OpsVars,
            name,
            lambda self, _op=name: self._check_nin(2).make_node(_op, *self.vars_),
        )

    flexible_ops = ["Squeeze"]
    for name in flexible_ops:
        if hasattr(OpsVars, name):
            continue
        setattr(
            OpsVars,
            name,
            lambda self, _op=name: self.make_node(_op, *self.vars_),
        )
