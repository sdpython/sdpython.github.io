from mlstatpy.nlp import CompletionSystem

mots = [
    "po",
    "po rouge",
    "po vert",
    "po orange",
    "port",
    "port blanc",
    "port bleu",
    "port rouge",
]
ens = CompletionSystem(mots)
ens.compute_metrics()
for el in ens:
    print("n={1} : M'={0} | {2}".format(el.mks1, el.weight, el.value))

n=0 : M'=1 | po
n=1 : M'=2 | po rouge
n=2 : M'=3 | po vert
n=3 : M'=4 | po orange
n=4 : M'=3 | port
n=5 : M'=4 | port blanc
n=6 : M'=5 | port bleu
n=7 : M'=6 | port rouge
