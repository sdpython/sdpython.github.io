def BJKST(stream, epsilon):
    s = len(stream)
    a1, a2 = random.sample(range(1, p), 2)
    b1, b2 = random.sample(range(0, p), 2)

    def h1(x):
        return ((a1 * x + b1) % p) % s

    def h2(x):
        return ((a2 * x + b2) % p) % s

    c = 0
    B = set()
    B_max = 1.0 / epsilon**2
    for x in stream:
        y = h1(x)
        k = mod_37bit_position[(-y & y) % 37]
        if k >= c:
            z = h2(x)
            B.add((z, k))
            while len(B) >= B_max:
                c += 1
                for z, k in B.copy():
                    if k < c:
                        B.remove((z, k))
    return 2**c * len(B)

def BJKST(stream, B_max, h1, h2):
    c = 0
    B = set()
    R = []
    removed = 0
    for x in stream:
        y = h1(x)
        k = mod_37bit_position[(-y & y) % 37]
        if k >= c:
            z = h2(x)
            B.add((z, k))
            while len(B) >= B_max:
                c += 1
                for z, k in B.copy():
                    if k < c:
                        B.remove((z, k))
                        removed += 1
        R.append([removed, len(B), c])
    return numpy.array(R)

def BJKST(stream, epsilon):
    a1, a2 = random.sample(range(1, p), 2)
    b1, b2 = random.sample(range(0, p), 2)
    # taille de la valeur de hashage dÃ©pend de la precision
    b = int(numpy.log(n) / epsilon**2)

    def h1(x):
        return ((a1 * x + b1) % p) % n

                c += 1
                for z, k in B.copy():
