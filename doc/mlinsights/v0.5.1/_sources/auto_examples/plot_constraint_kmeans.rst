Constraint KMeans

Simple example to show how to cluster keeping

    from collections import Counter

    import matplotlib.pyplot as plt
    import numpy
    from mlinsights.mlmodel import ConstraintKMeans
    from sklearn.cluster import KMeans
    from sklearn.datasets import make_blobs

    n_samples = 100
    data = make_blobs(
        n_samples=n_samples,
        n_features=2,
        centers=2,
        cluster_std=1.0,
        center_box=(-10.0, 0.0),
        shuffle=True,
        random_state=2,
    )
    X1 = data[0]
    data = make_blobs(
        n_samples=n_samples // 2,
        n_features=2,
        centers=2,
        cluster_std=1.0,
        center_box=(0.0, 10.0),
        shuffle=True,
        random_state=2,
    )
    X2 = data[0]

    X = numpy.vstack([X1, X2])
